NatCorder.CommitFrame

public static void CommitFrame (Frame frame);

Description

Commit a frame for encoding.

Note that all frames that have been previously acquired must be committed. Also note that a frame must not be committed more than once. Doing so will lead to undefined behaviour and possibly a crash.

When blitting to the frame, it is safest to do so in OnPostRender, OnRenderImage, or at the end of an application frame. This is because of platform-specific quirks. For example, on iOS Metal, blitting in Update will result in a black frame.

Some platforms (like WebGL) are immediate-encode platforms, meaning that frames are encoded immediately they are committed. As a result, timestamps on frames and audio sample buffers are wholly ignored. 

public class RecordCamera : MonoBehaviour {

    WebCamTexture cameraTexture;
       
    IEnumerator Start () {
        // Start camera
        cameraTexture = new WebCamTexture();
        cameraTexture.Play();
        // Start recording
        NatCorder.StartRecording(Configuration.Default, OnVideo);
        // Wait a while
        yield return new WaitForSeconds(10f);
        // Stop recording // The `OnVideo` callback will be invoked with the path to the recorded video
        NatCorder.StopRecording();
    }
    
    void Update () {
        if (NatCorder.IsRecording && cameraTexture.didUpdateThisFrame) {
            // Acquire an encoder frame from NatCorder
            var frame = NatCorder.AcquireFrame();
            // Blit the current camera preview frame to the encoder frame
            Graphics.Blit(cameraTexture, frame);
            // Commit the frame to NatCorder for encoding
            NatCorder.CommitFrame(frame);
        }
    }
    
    void OnVideo (string path) {
        // Do stuff with video
        ...
    }
}
